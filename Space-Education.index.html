            formatTime(timestamp) {

                const now = Date.now();

                const diff = now - timestamp;

                const minutes = Math.floor(diff / 60000);

                

                if (minutes < 1) return 'Just now';

                if (minutes < 60) return `${minutes}m ago`;

                const hours = Math.floor(minutes / 60);

                if (hours < 24) return `${hours}h ago`;

                const days = Math.floor(hours / 24);

                return `${days}d ago`;

            }

            trackInteraction(type, data = {}) {

                    this.planetsExplored.add(data.planet);

                }

                

                if (type === 'mission_completed') {

                    this.missionsCompleted++;

                    this.addXP(50, 'Mission Completed');

                }

                

                if (type === 'quiz_answered') {

                    this.totalQuizQuestions++;

                    if (data.correct) {

                        this.addXP(25, 'Correct Answer');

            initializeMouseControls() {

                    if (this.measureMode) return;

                    isDragging = true;

                    lastX = e.clientX;

                    lastY = e.clientY;

                });

 

                document.addEventListener('mousemove', (e) => {

                    if (!isDragging || this.measureMode) return;

                    

                    const deltaX = e.clientX - lastX;

                    const deltaY = e.clientY - lastY;

                    

                    this.rotation.y += deltaX * 0.5;

                    this.rotation.x -= deltaY * 0.5;

                    

                    this.updateSceneTransform();

                    

                    lastX = e.clientX;

                    lastY = e.clientY;

                });

 

                document.addEventListener('mouseup', () => {

                    isDragging = false;

                });

 

                this.scene.addEventListener('wheel', (e) => {

                    e.preventDefault();

                    const zoomFactor = e.deltaY > 0 ? 0.9 : 1.1;

                    this.zoom(zoomFactor);

                });

            }

                document.getElementById('earthMarsDist').textContent = `${Math.round(marsDist / 1000000)}M km`;

                document.getElementById('lunarPhase').textContent = phases[phaseIndex];

            }

                        }

                    }

                });

                

                this.updateMissionDisplay();

            }
